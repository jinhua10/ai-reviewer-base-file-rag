## 📚 组件使用示例

### Layout 布局
```jsx
import { Layout, Header, Footer, Sidebar } from '@components/layout'

<Layout
  header={<Header activeKey="qa" onMenuClick={handleClick} />}
  footer={<Footer />}
>
  <YourContent />
</Layout>
```

### Loading 加载
```jsx
import { Loading } from '@components/common'

// 全屏加载
<Loading spinning={true} fullscreen={true} />

// 包裹内容
<Loading spinning={isLoading}>
  <YourContent />
</Loading>
```

### Toast 提示
```jsx
import { Toast } from '@components/common'

Toast.success('操作成功')
Toast.error('操作失败')
Toast.loading('加载中...')
```

### Modal 对话框
```jsx
import { Modal } from '@components/common'

<Modal
  visible={visible}
  title="确认删除"
  onOk={handleOk}
  onCancel={handleCancel}
>
  <p>确定要删除吗？</p>
</Modal>

// 或使用方法调用
Modal.confirm({
  title: '确认删除',
  content: '确定要删除吗？',
  onOk: handleOk,
})
```

### Pagination 分页
```jsx
import { Pagination } from '@components/common'

<Pagination
  current={page}
  total={total}
  pageSize={10}
  onChange={handlePageChange}
/>
```

### Skeleton 骨架屏
```jsx
import { Skeleton } from '@components/common'

<Skeleton loading={isLoading} type="card">
  <YourContent />
</Skeleton>
```
